import { create } from 'zustand'

export type Tool = 'cursor' | 'miner' | 'belt' | 'splitter' | 'combiner' | 'blender' | 'sand-processor' | 'slot-machine' | 'deleter'
export type Direction = 0 | 1 | 2 | 3 // 0: N, 1: E, 2: S, 3: W

interface Notification {
  id: string
  text: string
  x: number
  y: number
  opacity: number
}

interface StoreState {
  currency: number
  addCurrency: (amount: number) => void
  tool: Tool
  setTool: (t: Tool) => void
  direction: Direction
  setDirection: (d: Direction) => void
  selectedEntityId: string | null
  setSelectedEntityId: (id: string | null) => void
  notifications: Notification[]
  addNotification: (text: string, x: number, y: number) => void
  removeNotification: (id: string) => void
}

import { persist } from 'zustand/middleware'

export const useStore = create<StoreState>()(
  persist(
    (set) => ({
      currency: 0,
      addCurrency: (amount) => set((state) => ({ currency: state.currency + amount })),
  tool: 'cursor',
  setTool: (tool) => set({ tool }),
  direction: 1,
  setDirection: (direction) => set({ direction }),
  selectedEntityId: null,
  setSelectedEntityId: (id) => set({ selectedEntityId: id }),
  notifications: [],
  addNotification: (text, x, y) => {
    const id = Math.random().toString(36).substr(2, 9)
    set((state) => ({
      notifications: [...state.notifications, { id, text, x, y, opacity: 1 }]
    }))
    setTimeout(() => {
      set((state) => ({
        notifications: state.notifications.filter(n => n.id !== id)
      }))
    }, 1500)
  },
  removeNotification: (id) => set((state) => ({
    notifications: state.notifications.filter(n => n.id !== id)
  }))
}), { name: 'store-save' }))
