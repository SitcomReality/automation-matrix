import { useRef, useEffect } from 'react';
import { useStore } from '../store';

export function ClickerMatrix() {
  const { addCurrency } = useStore();
  const cells = useRef<{active: boolean, timer: number}[]>(Array(25).fill(0).map(() => ({active: false, timer: 0})));
  const containerRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    let frame: number;
    const loop = () => {
      let needsRender = false;
      cells.current.forEach(c => {
        if (c.active) {
          c.timer--;
          if (c.timer <= 0) {
            c.active = false;
            needsRender = true;
          }
        }
      });
      if (needsRender && containerRef.current) {
        // manually update DOM classes for performance to avoid React re-renders on every frame
        const children = containerRef.current.children;
        for (let i = 0; i < 25; i++) {
          if (!cells.current[i].active) {
            children[i].classList.remove('bg-[#E94560]');
            children[i].classList.add('bg-[#1F2833]');
          }
        }
      }
      frame = requestAnimationFrame(loop);
    };
    frame = requestAnimationFrame(loop);
    return () => cancelAnimationFrame(frame);
  }, []);

  const handleEnter = (i: number, e: React.MouseEvent) => {
    const c = cells.current[i];
    if (!c.active) {
      c.active = true;
      c.timer = 120; // 2 seconds
      addCurrency(1);
      const target = e.target as HTMLDivElement;
      target.classList.remove('bg-[#1F2833]');
      target.classList.add('bg-[#E94560]');
    }
  };

  return (
    <div className="flex flex-col gap-1 p-2 bg-[#0B0C10] border-2 border-[#1F2833] rounded-sm">
      <div className="text-xs text-[#C5C6C7] font-bold tracking-tight">DOPAMINE MATRIX</div>
      <div ref={containerRef} className="grid grid-cols-5 gap-1">
        {Array(25).fill(0).map((_, i) => (
          <div
            key={i}
            onMouseEnter={(e) => handleEnter(i, e)}
            className="w-6 h-6 bg-[#1F2833] transition-colors duration-75 rounded-sm"
          />
        ))}
      </div>
    </div>
  );
}
